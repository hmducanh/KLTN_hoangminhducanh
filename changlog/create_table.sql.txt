CREATE TABLE `world`.`employee` (
  `ID` INT NOT NULL,
  `Username` VARCHAR(45) NULL,
  `Password` VARCHAR(45) NULL,
  `Role` INT NULL,
  `Fullname` VARCHAR(225) NULL,
  `Avatar` VARCHAR(45) NULL,
  `Age` INT NULL,
  PRIMARY KEY (`ID`))
ENGINE = InnoDB;


CREATE TABLE `world`.`educational_facility` (
  `ID` INT NOT NULL,
  `Name` VARCHAR(225) NULL,
  PRIMARY KEY (`ID`));

CREATE TABLE `world`.`criterion` (
  `ID` INT NOT NULL,
  `Description` TEXT NULL COMMENT 'Mô tả tiêu chí',
  PRIMARY KEY (`ID`));

  `CustomCriterionID13` INT NULL,
  PRIMARY KEY (`ID`));
